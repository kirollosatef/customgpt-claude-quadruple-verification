#!/bin/bash
# User Prompt Submit Hook
# Runs before git commits to enforce quality gates
#
# This integrates the engineering ratio check with the existing
# quality gate/critic agent system from customgpt-security-analyst

# Exit on error
set -e

# Configuration
HOOKS_DIR="$HOME/.claude/hooks"
ENGINEERING_RATIO_HOOK="$HOOKS_DIR/pre-commit-engineering-ratio"

# Colors for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# Check if this is a git commit operation
if echo "$CLAUDE_USER_PROMPT" | grep -qiE "(git commit|commit.*changes|create.*commit)"; then
    echo ""
    echo -e "${YELLOW} Running Quality Gate: Engineering Ratio Check${NC}"
    echo ""

    # Run the engineering ratio check
    if [ -x "$ENGINEERING_RATIO_HOOK" ]; then
        if "$ENGINEERING_RATIO_HOOK"; then
            echo -e "${GREEN} Quality gate passed: Engineering ratio is acceptable${NC}"
            echo ""
        else
            echo -e "${RED} Quality gate failed: Engineering ratio too low${NC}"
            echo ""
            echo "This commit would violate the engineering ratio target (Issue #49)."
            echo "Please focus on engineering work before committing more docs."
            echo ""
            exit 1
        fi
    else
        echo -e "${YELLOW}  Engineering ratio hook not found or not executable${NC}"
        echo "Skipping quality gate check..."
        echo ""
    fi
fi

# Allow the commit to proceed
exit 0
